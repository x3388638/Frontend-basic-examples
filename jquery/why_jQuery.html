Use pure JavaScript to show hidden block after the button.


Hidden block: 
document.getElementById('btn2').addEventListener('click', function (e) {
	var targetElement = Array.prototype.filter.call(e.target.parentNode.children, function (child) {
		return child !== e.target;
	}).filter(function (el, i) {
		if (el.tagName.toLowerCase() === 'pre') {
			return true;
		}

		return false;
	})[0];

	targetElement.style.display = '';
});
